  • to select DB: use dbname
  • to list DB: show databases
  • to create DB: create database dbname
  • to delete DB: drop database dbname
  • to show tables: show tables(select DB first)
  • to create table: create table tablename(colname datatype)
  • to delete table: drop table tablename
  • to insert into table: insert into tablename(colname) values(val)
  • log commands: tee log.txt
  • stop logging: notee
