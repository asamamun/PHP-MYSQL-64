CREATE FUNCTION catname(cat_id INT)
RETURNS VARCHAR(60)
DETERMINISTIC
BEGIN
  DECLARE cat_name VARCHAR(60);

  SELECT name INTO cat_name FROM categories  WHERE id = cat_id  LIMIT 1;

  RETURN cat_name;
END//
